cmake_minimum_required(VERSION 3.13)

set(ENV{PICO_SDK_PATH} ${CMAKE_CURRENT_LIST_DIR}/libs/pico-sdk)
include($ENV{PICO_SDK_PATH}/external/pico_sdk_import.cmake)

project(my_project C CXX ASM)
set(CMAKE_C_STANDARD 11)
set(CMAKE_CXX_STANDARD 17)

pico_sdk_init()
add_subdirectory(${CMAKE_CURRENT_LIST_DIR}/libs/freemodbus-rp2040/demo/RP2040/)

add_executable(my_project
    main.c
)

target_link_libraries(my_project
    pico_stdlib
)

add_compile_options(-Wall
    -Werror
    -Wno-format
    -Wno-unused-function
    -Wno-maybe-uninitialized
)

pico_add_extra_outputs(my_project)

target_include_directories(freemodbus-rp2040 PUBLIC
    ${CMAKE_CURRENT_LIST_DIR}/freemodbus_pinmap/
)

target_link_libraries(my_project
    freemodbus-rp2040
)
